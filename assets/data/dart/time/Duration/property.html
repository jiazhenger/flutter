# 定义 Duration A = Duration(days: 1, hours: 1, minutes: 1, seconds: 1, milliseconds: 1, microseconds: 1); // => 25:01:01.001001 # 获取具体时间数 int days = A.inDays; // 天数 => 1 int hours = A.inHours; // 小时数 => 25 int minutes = A.inMinutes; // 分钟数 => 1501 int seconds = A.inSeconds; // 秒数 => 90061 int milliseconds = A.inMilliseconds; // 毫秒数 => 90061001 int microseconds = A.inMicroseconds; // 微秒数 => 90061001001
