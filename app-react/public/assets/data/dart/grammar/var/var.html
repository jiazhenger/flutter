// 声明任意变量 （不能带数据类型） var a; // 声明一个变量 var a,b,c; // 声明多个变量 var a = 0; // 声明并初始化一个变量 var a = 0,b=1 ; // 声明并初始化多个变量 var a = 0,b,c ; // 混合声明 // 声明可改变一次的变量（必须初始化变量） final a = 0; final int a = 0; // 声明数据类型变量 final Map<String,String> a = {}; // 声明泛类型变量 // 声明常量，不可改变（必须初始化变量） const a = 0; const int a = 0; // 声明数据类型常量 const Map<String,String> a = {}; // 声明泛类型变量 static const int a = 0; // 在 class 类中声明必须使用 static
