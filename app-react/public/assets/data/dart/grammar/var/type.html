// 数据类型声明变量，可改变多次，与var声明用法相同 int a = 0; // 整数 double a = 1.2; // 小数 num a = 10; // 数字 String a = 'string' ; // 字符串 bool a = true ; // 布尔值 Map a = {}; // 映射 Map<String,String> a = {'a':'1'}; // 映射泛类型 Set a = {0,1,2}; // 集合 Set <String> set = {'0','1','2'} // 集合泛类型 List a = [1,2,3]; // 列表 List <num> a = [1,2,3]; // 列表泛类型 dynamic a = 0; // 无类型，可以声明任意数据类型变量，不能加泛类型 Object a = []; // 通用类型，可以声明任意数据类型变量 Object a = <String>[]; // 通用类型泛类型
